// --- Filtro de Élite (Chat cualificador) ---
const EliteFilter = () => {
  const [step, setStep] = useState(0);
  const [chat, setChat] = useState([]);
  const [loading, setLoading] = useState(false);

  const steps = [
    { q: "¿En qué área se enfoca su proyecto?", opts: ["Mobiliario a Medida", "Shop Fitting / Retail", "Interiorismo Técnico"] },
    { q: "¿Cuál es el rango de presupuesto estimado?", opts: ["5k€ - 15k€", "15k€ - 50k€", "Proyectos > 50k€"] },
    { q: "Entendido. ¿Para cuándo requiere la entrega?", opts: ["Urgente (< 1 mes)", "Plazo estándar (3-6 meses)", "Planificación futura"] },
  ];

  const handleSelect = (opt) => {
    setChat([...chat, { type: 'user', text: opt }]);
    setLoading(true);
    setTimeout(() => {
      setLoading(false);
      setStep(step + 1);
    }, 800);
  };

  return (
    <section id="filter" className="py-32 px-6 md:px-12 bg-[#0F0F10] border-t border-[#2F3236]">
      <div className="max-w-7xl mx-auto grid grid-cols-1 lg:grid-cols-2 gap-20 items-center">
        <div>
          <SectionTitle 
            number="02" 
            title="Cualificación de Proyectos" 
            subtitle="Filtramos cada solicitud para asegurar que el nivel de exigencia técnica coincide con nuestra capacidad de producción premium."
          />
          <div className="space-y-4 font-mono text-[10px] text-[#7E848B]">
            <div className="flex items-center gap-3">
              <Cpu className="w-4 h-4 text-[#4A90E2]" /> STATUS: OPERATIVO
            </div>
            <div className="flex items-center gap-3">
              <Compass className="w-4 h-4 text-[#4A90E2]" /> MODO: FILTRADO DE ÉLITE
            </div>
          </div>
        </div>

        <div className="bg-[#1A1A1C] border border-[#2F3236] p-8 relative overflow-hidden min-h-[450px] flex flex-col justify-between">
          <div className="space-y-6 overflow-y-auto max-h-[300px] pr-2">
            <div className="bg-[#2F3236] p-4 text-[13px] text-white/90 font-light max-w-[85%] self-start">
              Bienvenido al sistema de acceso Z_DESIGN96. Para optimizar nuestra consultoría, defina brevemente su necesidad.
            </div>
            
            {chat.map((msg, i) => (
              <div key={i} className={`p-4 text-[13px] font-light max-w-[85%] ${msg.type === 'user' ? 'bg-[#4A90E2] text-white ml-auto' : 'bg-[#2F3236] text-white/90'}`}>
                {msg.text}
              </div>
            ))}

            {loading && <div className="text-[10px] font-mono text-[#4A90E2] animate-pulse">ANALIZANDO DATOS...</div>}

            {step < steps.length && !loading && (
              <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} className="space-y-4 pt-4">
                <div className="text-[12px] text-[#7E848B] font-mono uppercase mb-4">{steps[step].q}</div>
                <div className="flex flex-wrap gap-2">
                  {steps[step].opts.map((o, i) => (
                    <button 
                      key={i} 
                      onClick={() => handleSelect(o)}
                      className="border border-[#2F3236] px-4 py-2 text-[11px] font-mono text-white/60 hover:border-[#4A90E2] hover:text-[#4A90E2] transition-all"
                    >
                      {o}
                    </button>
                  ))}
                </div>
              </motion.div>
            )}

            {step === steps.length && (
              <motion.div initial={{ opacity: 0 }} animate={{ opacity: 1 }} className="bg-[#4A90E2]/10 border border-[#4A90E2] p-6 text-center">
                <div className="text-[#4A90E2] text-[12px] font-mono mb-2 uppercase tracking-widest">Aptitud Técnica Confirmada</div>
                <p className="text-white/80 text-[13px] font-light">Joan recibirá un briefing detallado. Nos pondremos en contacto en 24h.</p>
                <button className="mt-6 w-full py-3 bg-[#4A90E2] text-white text-[10px] font-mono uppercase tracking-widest hover:bg-white hover:text-[#0F0F10] transition-all">Finalizar Contacto</button>
              </motion.div>
            )}
          </div>

          <div className="mt-8 pt-4 border-t border-[#2F3236] flex justify-between items-center font-mono text-[8px] text-[#2F3236]">
            <span>SYSTEM_V.4.0</span>
            <span>ENCRYPTED_LINK_ACTIVE</span>
          </div>
        </div>
      </div>
    </section>
  );
};
